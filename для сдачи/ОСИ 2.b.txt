sys/syscall - импорт нужного номера сис кола (в х64 номер write = 1, в x86 - 4)
movq $1, %rdi - помещает номер файлового дескриптора в регистр rdi
movq %0, %%rsi - помещает указатель на начало строки в регистр %rsi
movq $14, %rdx - помещает длину строки (14) в регистр %rdx
movq %0, %%rax - помещает номер сискола в регистр %rax
syscall - вызов сискола, номер которого находится в регистре %rax, а аргументы в рассмотренных ранее регистрах
